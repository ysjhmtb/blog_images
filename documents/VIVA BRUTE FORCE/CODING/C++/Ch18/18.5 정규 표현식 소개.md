# 18.5 정규 표현식 소개



```C++

// main.cpp

#include <iostream>
#include <regex>

int main() {

    {
        // std::regex re("\\d+");                       // digit 한개 이상의 숫자.
        // std::regex re("[[:digit:]]{3}");             // digit 세개
        // std::regex re("[A-Z]+");                     // 대문자 한개 이상.
        //std::regex re("[A-Z]{1,5}");                  // 대문자 최소 하나 최대 다섯개
        std::regex re("([0-9]{1})([-]?)([0-9]{1,4})");  // 숫자 한개, - 있어도 되고 없어도 되고, 숫자 1~4

        std::string str;

        while (true) {
            std::getline(std::cin, str);

            if (std::regex_match(str, re))
                std::cout << "Match " << std::endl;
            else
                std::cout << "No match " << std::endl;

            // print matches
            // https://en.cppreference.com/w/cpp/regex/regex_iterator
            {
                auto begin = std::sregex_iterator(str.begin(), str.end(), re);
                auto end = std::sregex_iterator();
                for (auto itr = begin; itr != end; ++itr) {
                    std::smatch match = *itr;
                    std::cout << match.str() << " ";
                }
                std::cout << std::endl;
            }
        }
    }


    return 0;
}
```

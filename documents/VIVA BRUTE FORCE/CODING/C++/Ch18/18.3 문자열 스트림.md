# 18.3 문자열 스트림





```C++

// main.cpp

#include <iostream>
#include <sstream>


int main() {

    {
        std::stringstream os;

        os << "Hello, World!";          // << : insertion operator, >> : extraction operator
        // os.str("Hello, World!");
        os << "Hello, World!2";         // Hello, World!Hello, World!2
        os.str("Hello, World! \n");     // Hello, World! \n

        std::string str;
        // os >> str;   // 문자열 손실 발생
        str = os.str();

        std::cout << str << std::endl;

        std::cout << std::endl;

    }

    {

        std::stringstream os;

        os << "12345 67.89";
        std::string str1;
        std::string str2;

        os >> str1 >> str2; // 빈칸을 기점으로 갈린다.

        std::cout << str1 << " | " << str2 << std::endl;    // 12345 | 67.89

        std::cout << std::endl;

    }

    {
        std::stringstream os;

        int i = 12345;
        double d = 67.89;

        os << i << " " << d;

        std::string str1;
        std::string str2;

        os >> str1 >> str2;

        std::cout << str1 << " | " << str2 << std::endl;    // 12345 | 67.89

        std::cout << std::endl;
    }

    {
        std::stringstream os;

        int i = 12345;
        double d = 67.89;

        os << i << " " << d;

        int i2;
        double d2;

        os >> i2 >> d2;

        std::cout << i2 << " | " << d2 << std::endl;     // 12345 | 67.89

        std::cout << std::endl;
    }

    {
        std::stringstream os;

        os << "12345 67.89";

        int i2;
        double d2;

        os >> i2 >> d2;

        std::cout << i2 << " | " << d2 << std::endl;     // 12345 | 67.89

        std::cout << std::endl;
    }

    {
        std::stringstream os;
        os << "12345 67.89";

        os.str("");
        os.clear();

        std::cout << os.str() << std::endl; // 공백

        std::cout << std::endl;
    }


    return 0;
}
```

# 18.1 istream으로 입력받기



```C++

// main.cpp

#include <iostream>
#include <string>
#include <iomanip>  // input/output manipulators

int main() {

    {
        std::cout << "Enter a number" << std::endl;

        int i;
        std::cin >> i;

        std::cout << i << std::endl;
        std::cout << std::endl;
    }

    {
        char buf[3];
        std::cout << "자동으로 세 글자 한정 입력" << std::endl;

        std::cin >> std::setw(3) >> buf;    // null character 포함 세 글자.
        // 이 시점에서 6글자를 입력하면 아래에서 연속적으로 입력받게 된다.
        std::cout << buf << std::endl;

        std::cin >> std::setw(3) >> buf;
        std::cout << buf << std::endl;

        std::cin >> std::setw(3) >> buf;
        std::cout << buf << std::endl;

        std::cout << std::endl;
    }

    {
        char ch;
        while (std::cin >> ch)  // 입력한 빈칸이 사라져서 출력된다.
            std::cout << ch;

        std::cout << std::endl;
    }

    {
        char ch;
        while (std::cin.get(ch))
            std::cout << ch;

        std::cout << std::endl;
    }

    {
        char buf[5];
        std::cout << "std::cin.get(buf, 5)" << std::endl;
        std::cin.get(buf, 5);
        std::cout << buf << std::endl;

        std::cout << std::endl;

    }

    {
        char buf[5];
        std::cout << "std::cin.gcount()" << std::endl;
        std::cin.get(buf, 5);
        std::cout << std::cin.gcount() << " " << buf << std::endl;

        std::cin.get(buf, 5);
        std::cout << buf << std::endl;

        std::cout << std::endl;

    }

    {
        char buf[100];
        std::cout << "getline" << std::endl;
        std::cin.getline(buf, 100); // getline은 개행문자까지 읽어들인다.
        std::cout << std::cin.gcount() << " " << buf << std::endl;

        std::cin.getline(buf, 100);
        std::cout << buf << std::endl;

        std::cout << std::endl;

    }

    {
        std::string buf;
        std::cout << "std::string !!!! " << std::endl;

        std::getline(std::cin, buf);
        std::cout << std::cin.gcount() << " " << buf << std::endl;

        std::getline(std::cin, buf);
        std::cout << buf << std::endl;

        std::cout << std::endl;

    }

    {
        char buf[1024];
        std::cin.ignore(2); // 입력받을 때 처음 두 글자 무시.
        std::cin >> buf;
        std::cout << buf << std::endl;
    }

    {
        char buf[1024];

        std::cout << (char) std::cin.peek() << std::endl;   // 버퍼에서 꺼내지는 않고 첫 한 글자 보기.

        std::cin >> buf;
        std::cout << buf << std::endl;

        std::cin.unget();   // 버퍼에 마지막 글자 하나가 남게 한다.
        std::cin.putback('A');  // 버퍼에 글자 넣기.

        std::cin >> buf;
        std::cout << buf << std::endl;
    }

    {
        int i;
        float f;

        std::cin >> i >> f;
        std::cout << i << " " << f;

        std::cout << std::endl;
    }


    return 0;
}
```

# 12.2 가상 함수와 다형성



```C++

// main.cpp

#include <iostream>
#include <string>

using namespace std;


class A {
public:
    virtual void print() { cout << "A" << endl; }
};

class B : public A {
public:
    void print() { cout << "B" << endl; }
};

class C : public B {
public:
    void print() { cout << "C" << endl; }
};

class D : public C {
public:
    void print() { cout << "D" << endl; }
};


int main() {

    A a;
    B b;
    C c;
    D d;

    // A &ref = b;
    // ref.print();    // A, A 클래스의 print 함수가 virtual이 아닌 경우.

    B ref = c;
    ref.print();    // B, A에는 virtual이 있고, B에는 없다.


    return 0;
}

```

# 12.6 가상 (함수) 표



```C++

// main.cpp

#include <iostream>
#include <string>

using namespace std;

class Base {
public:
    // FunctionalPointer *__vptr;
    virtual void fun1() {};

    virtual void fun2() {};
};

class Der : public Base {
public:
    // FunctionalPointer *__vptr;
    virtual void fun1() {};

    virtual void fun3() {};

};

int main() {

    cout << sizeof(Base) << endl;   // 결과에 대해 추가 학습 필요.
    cout << sizeof(Der) << endl;

    return 0;
}

```

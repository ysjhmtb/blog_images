# 11.4 유도된 클래스들의 생성과 초기화



```C++

// main.cpp

#include <iostream>


class Mother {
private:
    int m_i;

public:
    Mother(const int &i_in = 0) : m_i(i_in) {
        std::cout << "Mother construction" << std::endl;
    }
};

class Child : public Mother {
private:
    double m_d;

public:
    Child() : Mother(1024), m_d(1.0) {
        std::cout << "Child construction" << std::endl;
    }
};


class A {
public:

    A(int a) {
        std::cout << "A: " << a << std::endl;
    }

    ~A() {
        std::cout << "Destructor A" << std::endl;
    }

};

class B : public A {

public:
    B(int a, double b)
            : A(a) {
        std::cout << "B: " << b << std::endl;
    }

    ~B() {
        std::cout << "Destructor B" << std::endl;
    }

};

class C : public B {

public:
    C(int a, double b, char c)
            : B(a, b) {
        std::cout << "C: " << c << std::endl;
    }

    ~C() {
        std::cout << "Destructor C" << std::endl;
    }

};


int main() {


//    Mother construction
//    Child construction
    Child c1;


    std::cout << sizeof(Mother) << std::endl;
    std::cout << sizeof(Child) << std::endl;


    C c(1024, 3.14, 'c');


    return 0;
}
```

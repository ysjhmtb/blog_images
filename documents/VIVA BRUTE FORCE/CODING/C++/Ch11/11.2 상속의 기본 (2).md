```C++

// Person.h

#ifndef DEC_PERSON_H
#define DEC_PERSON_H

#include <iostream>
#include <string>

class Person {
private:
    std::string m_name;

public:
    Person(const std::string &name_in = "No Name")
            : m_name(name_in) {

    }

    void setName(const std::string &name_in) {
        m_name = name_in;
    }

    std::string getName() const {
        return m_name;
    }

    void doNothing() const {
        std::cout << m_name << "  is doing nothing" << std::endl;
    }


};

#endif //DEC_PERSON_H

```

```c++
// Teacher.h

#ifndef DEC_TEACHER_H
#define DEC_TEACHER_H

#include "Person.h"

class Teacher : public Person {

private:


public:
    Teacher(const std::string &name_in = "No Name")
            : Person(name_in) {

    }

    void teach() {
        std::cout << getName() << " is teaching" << std::endl;
    }

    friend std::ostream &operator<<(std::ostream &out, const Teacher &teacher) {

        out << teacher.getName();
        return out;

    }


};

#endif //DEC_TEACHER_H
```

```C++

// Student.h

#ifndef DEC_STUDENT_H
#define DEC_STUDENT_H

#include "Person.h"

class Student : public Person {

private:
    int m_intel;

public:
    Student(const std::string &name_in = "No Name", const int &intel_in = 0)
            : Person(name_in), m_intel(intel_in) {

    }


    void setIntel(const int &intel_in) {
        m_intel = intel_in;
    }

    int getIntel() {
        return m_intel;
    }

    void study() {
        std::cout << getName() << " is studying" << std::endl;
    }

    friend std::ostream &operator<<(std::ostream &out, const Student &student) {
        out << student.getName() << " " << student.m_intel;
        return out;
    }


};

#endif //DEC_STUDENT_H

```

```C++

// main.cpp

#include "Student.h"
#include "Teacher.h"

using namespace std;

int main() {

    Student std("Jack Jack");
    std.setName("Jack Jack 2");
    cout << std.getName() << endl;

    Teacher teacher1("Dr. H");
    teacher1.setName("Dr. K");
    cout << teacher1.getName() << endl;

    std.doNothing();
    teacher1.doNothing();

    std.study();
    teacher1.teach();

    return 0;
}
```

# 11.8 상속 받은 함수를 감추기



```C++

// main.cpp

#include <iostream>

using namespace std;


class Base {
protected:
    int m_i;

public:
    Base(int value)
            : m_i(value) {

    }

    void print() {
        cout << "I'm base " << endl;
    }


};

class Derived : public Base {

private:
    double m_d;

public:
    Derived(int value)
            : Base(value) {

    }

    using Base::m_i;

private:
    using Base::print; // do not add ()
    // using 대신에 void print() = delete; 가능.


};


int main() {


    Derived derived(7);

    derived.m_i = 1024; // using Base::m_i; 때문에 가능해졌다.
    // derived.print();    // error.
    // private:
    //     using Base::print;

    return 0;
}

```

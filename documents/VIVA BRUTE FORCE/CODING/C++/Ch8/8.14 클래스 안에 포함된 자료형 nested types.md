# 8.14 클래스 안에 포함된 자료형 nested types



```C++
#include <iostream>

using namespace std;



class Fruit {

public:
    // class Fruit 내부 사용 목적
    enum class FruitType {
        APPLE, BANANA, CHERRY,
    };

private:
    FruitType m_type;

public:
    Fruit(FruitType type) : m_type(type) {};

    FruitType getType() { return m_type; }
};


int main() {

    // enum class FruitType -> Fruit::FruitType::APPLE
    // enum FruitType       -> Fruit::APPLE
    Fruit apple(Fruit::FruitType::APPLE);

    if (apple.getType() == Fruit::FruitType::APPLE) {
        cout << "Apple" << endl;
    }


    return 0;
}
```

# 8.12 친구 함수와 클래스 friend



```C++
#include <iostream>

using namespace std;

// 전방 선언 forward declaration.
// class B에서 void doSomething(A &a); 사용을 위해 필요.

class A;

class B {
private:
    int m_value = 2;

public:
    // class A의 정의가 아래에 위치해 있다. 따라서 
    // cout << a.m_value << endl; 의 사용이 불가능하다.
    // 선언과 정의의 분리 이유.
    void doSomething(A &a);
};


class A {
private:
    int m_value = 1;

    friend void B::doSomething(A &a);

};


void B::doSomething(A &a) {
    cout << a.m_value << endl;
}


int main() {

    A a;
    B b;
    b.doSomething(a);

    return 0;
}
```

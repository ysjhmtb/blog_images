# 9.9 복사 생성자, 복사 초기화 반환값 최적화



```C++

#include <iostream>
#include <cassert>

using namespace std;


class Fraction {
private:
    int m_numerator;
    int m_denominator;

public:
    Fraction(int num = 0, int den = 1)
            : m_numerator(num), m_denominator(den) {
        assert(den != 0);
    }

    Fraction(const Fraction &fraction) // copy cunstructor
            : m_numerator(fraction.m_numerator), m_denominator(fraction.m_denominator) {
        cout << "Copy constructor called" << endl;
    }

    friend std::ostream &operator<<(std::ostream &out, const Fraction &f) {
        cout << f.m_numerator << " / " << f.m_denominator << endl;
        return out;
    }
};

Fraction doSomething() {
    Fraction temp(1, 2);
    return temp;
}


int main() {

    Fraction frac(3, 5);
    Fraction fr_copy(frac);

    // 이 경우에는 복사 생성자가 호출되지 않는다. 컴파일러가  Fraction fr_copy(3, 10);로 바꾼다.
    // Fraction fr_copy(Fraction(3, 10));


    cout << frac << " " << fr_copy << endl;

    Fraction result = doSomething();
    cout << result << endl;

    return 0;
}
```

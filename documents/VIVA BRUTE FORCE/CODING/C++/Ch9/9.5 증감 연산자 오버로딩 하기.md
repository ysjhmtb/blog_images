# 9.5 증감 연산자 오버로딩 하기



```C++
#include <iostream>
#include <fstream>
#include <vector>
#include <algorithm>

using namespace std;

class Digit {
private:
    int m_digit;

public:
    Digit(int digit = 0) : m_digit(digit) {}

    // prefix
    Digit &operator++() {
        ++m_digit;
        return *this;
    }

    // postfix
    // Digit operator++(int) 이 아니라 실수로 Digit &operator++(int)
    // 사용하였더니
    // arning: reference to stack memory associated with local variable 'temp' returned [-Wreturn-stack-address]
    //        return temp;
    // 발생
    Digit operator++(int) {
        Digit temp(m_digit);
        // 위에서 오버로딩 한 연산자 활용
        ++(*this);
        return temp;
    }


    friend ostream &operator<<(ostream &out, const Digit &d) {
        out << d.m_digit;
        return out;
    }

};


int main() {

    Digit d(5);
    cout << ++d << endl;
    cout << d++ << endl;
    cout << d << endl;


    return 0;
}
```

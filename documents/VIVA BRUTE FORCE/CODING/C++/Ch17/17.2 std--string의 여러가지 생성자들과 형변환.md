# 17.2 std::string의 여러가지 생성자들과 형변환



```C++

// main.cpp

#include <iostream>
#include <string>
#include <vector>
#include <sstream>


template<typename T>
std::string ToString(T x) {
    std::ostringstream osstream;
    osstream << x;
    return osstream.str();
}

template<typename T>
bool FromString(const std::string &str, T &x) {
    std::istringstream isstream(str);
    return (isstream >> x) ? true : false;
}


int main() {

    {
        std::string my_string("my string");
        std::string second_string(my_string);
        std::cout << second_string << std::endl;
        std::cout << std::endl;

    }

    {
        std::string my_string("my string");
        std::string second_string(my_string, 3);
        std::cout << second_string << std::endl;
        std::cout << std::endl;

    }

    {
        std::string my_string("my string");
        std::string second_string(my_string, 3, 5);
        std::cout << second_string << std::endl;
        std::cout << std::endl;
    }

    {
        const char *my_string = "my string";
        std::string second_string(my_string, 3, 5);
        std::cout << second_string << std::endl;
        std::cout << std::endl;
    }

    {
        std::string second_string(10, 'A');
        std::cout << second_string << std::endl;
        std::cout << std::endl;
    }

    {
        std::vector<char> vec;

        for (auto e:"Today is a good day.")
            vec.push_back(e);

        std::string second_string(vec.begin(), vec.end());
        std::cout << second_string << std::endl;
        std::cout << std::endl;
    }

    {
        std::vector<char> vec;

        for (auto e:"Today is a good day.")
            vec.push_back(e);

        std::string second_string(vec.begin(), std::find(vec.begin(), vec.end(), 'g'));
        std::cout << second_string << std::endl;
        std::cout << std::endl;
    }

    {
        std::string my_str(std::to_string(1004));
        my_str += std::to_string(128);
        std::cout << my_str << std::endl;
        std::cout << std::endl;
    }

    {
        std::string my_str(std::to_string(1004));
        int i = std::stoi(my_str);
        std::cout << my_str << std::endl;
        std::cout << i << std::endl;
        std::cout << std::endl;
    }

    {
        std::string my_str(ToString(3.141592));
        std::cout << my_str << std::endl;
        std::cout << std::endl;
    }

    {
        std::string my_string(ToString(3.141592));
        double d;

        if (FromString(my_string, d))
            std::cout << d << std::endl;
        else
            std::cout << "Cannot convert string to double" << std::endl;

        std::cout << std::endl;
    }


    return 0;
}
```

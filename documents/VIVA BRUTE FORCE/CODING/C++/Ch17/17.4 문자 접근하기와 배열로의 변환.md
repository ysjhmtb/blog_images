# 17.4 문자 접근하기와 배열로의 변환





```C++

// main.cpp

#include <iostream>
#include <string>
#include <vector>


int main() {

    {
        std::string my_str("abcdefg");

        std::cout << my_str[0] << std::endl;
        std::cout << my_str[3] << std::endl;

        my_str[3] = 'Z';
        std::cout << my_str << std::endl;

        try {
            // my_str[100] = 'X';   // 여기서는 throw 안해 준다.
            my_str.at(100) = 'X';
        } catch (std::exception &e) {
            std::cout << e.what() << std::endl;
        }

        std::cout << std::endl;

    }

    {
        std::string my_str("abcdefg");  // null character가 존재하지 않는다.
        std::cout << my_str.c_str() << std::endl;

        const char *arr = my_str.c_str();   // null character가 붙는다.
        std::cout << (int) arr[6] << std::endl;
        std::cout << (int) arr[7] << std::endl;

        std::cout << std::endl;
    }

    {
        std::string my_str("abcdefg");
        std::cout << my_str.c_str() << std::endl;

        const char *arr = my_str.data();   // c_str()과 같다.
        std::cout << (int) arr[6] << std::endl;
        std::cout << (int) arr[7] << std::endl;

        std::cout << std::endl;
    }

    {
        std::string my_str("abcdefg");
        char buf[20];

        my_str.copy(buf, 5, 0); // offset 0. null character는 자동으로 넣어주지 않는다.

        buf[5] = '\0';

        std::cout << buf << std::endl;


    }


    return 0;
}



```

# 17.3 std::string의 길이와 용량



```C++

// main.cpp

#include <iostream>
#include <string>
#include <vector>
#include <sstream>


int main() {

    {
        std::string my_str("012345678");
        std::cout << my_str.size() << std::endl;
        std::cout << std::boolalpha;
        std::cout << my_str.empty() << std::endl;   // false
        std::cout << std::endl;
    }

    {
        std::string my_str("");
        std::cout << std::boolalpha;
        std::cout << my_str.empty() << std::endl;   // true, null character를 포함해서 판단하지 않는다.
        std::cout << std::endl;
    }

    {
        std::string my_str("01234567");

        std::cout << std::boolalpha;
        std::cout << my_str.length() << std::endl;
        std::cout << my_str.size() << std::endl;
        std::cout << my_str.capacity() << std::endl;    // OS also wants not to repeat new and delete for speed.
        std::cout << my_str.max_size() << std::endl;
        std::cout << std::endl;
    }

    {
        std::string my_str("01234567");
        my_str.reserve(1000);   // request to reserve at least size

        std::cout << my_str.length() << std::endl;
        std::cout << my_str.size() << std::endl;
        std::cout << my_str.capacity() << std::endl;
        std::cout << my_str.max_size() << std::endl;
        std::cout << std::endl;


        std::cout << std::boolalpha;
    }

    return 0;
}
```

# 7월 10일 면접 문제

<br>

7월 10일에 있었던 면접에서 면접 질의 이후 화이트보드에 다음과 같은 두 개의 코딩 문제를 풀었다. 당시에는 파이썬으로 풀었는데, 이번에는 C++로 작성해 보았다.

<br>

```C++
#include <iostream>
#include <string>
#include <iterator>
#include <vector>
#include <algorithm>
#include <sstream>
#include <map>

using namespace std;

/*
710 1)
블랭크로 구별되는 문자열을 입력받는다.
가장 많이 등장하는 상위 4개의 단어를 반환하라.

"단어1 단어2 단어3 단어4 단어5 단어6 단어7 단어8 단어9 단어1 단어2 단어3 단어4"


710 2)
오름차순으로 정렬되어 있고 크기가 같은 숫자 배열 3개를 입력받는다.
배열 세곳에서 모두 존재하는 숫자를 반환하라. 
시간복잡도는 O(N) 이하여야 하며, 해시맵을 사용해서는 안 된다.

{1, 2, 3, 4, 5}
{3, 4, 5, 6, 7}
{2, 3, 9, 10, 11}
 */

class Solution
{

public:
    vector<string> firstProblem(string input)
    {
        stringstream stream(input);
        istream_iterator<string> begin(stream);
        istream_iterator<string> end;
        vector<string> words(begin, end);

        map<string, int> wordCount;

        for (string word : words)
        {
            // clang++ -std=c++11 -g 710.cpp -o 710
            // cout << word << " ";

            if (wordCount.find(word) == wordCount.end())
            {
                wordCount[word] = 1;
            }
            else
            {
                wordCount[word] += 1;
            }
        }

        vector<string> tempStorage;

        for (map<string, int>::iterator it = wordCount.begin(); it != wordCount.end(); it++)
        {
            // cout << it->first << " : " << it->second << endl;
            tempStorage.push_back(it->first);
        }

        for (unsigned i = 0; i < tempStorage.size(); i++)
        {
            for (unsigned j = 0; j < tempStorage.size(); j++)
            {
                if (wordCount[tempStorage.at(i)] > wordCount[tempStorage[j]])
                {
                    string temp = tempStorage.at(i);
                    tempStorage.at(i) = tempStorage.at(j);
                    tempStorage.at(j) = temp;
                }
            }
        }

        vector<string> result;
        for (unsigned i = 0; i < 4; i++)
        {
            cout << tempStorage.at(i) << endl;
            result.push_back(tempStorage.at(i));
        }

        return result;
    }

    vector<int> secondProblem(vector<int> &a1, vector<int> &a2, vector<int> &a3)
    {
        vector<int> result;

        int i = 0, j = 0, k = 0;

        while (i < a1.size() && j < a2.size() && k < a3.size())
        {

            int maxNum = max(a1.at(i), max(a2.at(j), a3.at(k)));

            if (a1.at(i) == a2.at(j) && a2.at(j) == a3.at(k))
            {
                
                result.push_back(a1.at(i));
                i++;
                j++;
                k++;
            }
            else
            {

                if (a1.at(i) < maxNum)
                {
                    i++;
                }
                if (a2.at(j) < maxNum)
                {
                    j++;
                }
                if (a3.at(k) < maxNum)
                {
                    k++;
                }
            }
        }

        return result;
    }
};

int main()
{

    Solution ref;
    ref.firstProblem("단어1 단어2 단어3 단어4 단어5 단어6 단어7 단어8 단어9 단어1 단어2 단어3 단어4");

    vector<int> arr1 = {1, 2, 3, 4, 5};
    vector<int> arr2 = {3, 4, 5, 6, 7};
    vector<int> arr3 = {2, 3, 9, 10, 11};

    ref.secondProblem(arr1, arr2, arr3);

    return 0;
}
```
